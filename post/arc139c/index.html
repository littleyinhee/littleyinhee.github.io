int n,m;
int b[12]={0,0,0,0,4,7,10,5,8,11,6,9};
void Yorushika(){
	scanf("%d%d",&n,&m);
	bool fl=0;
	if(n<m){
		swap(n,m);
		fl=1;
	}
	vector<pii> ans;
	rep(i,4,m*3+n){
		int p=(i-4)%8+4;
		int &j=b[p];
		int x=((i+j)/4-(i-j)/2)/2,y=(i-x)/3;
		while(x<1||y>m){
			j+=8;
			x=((i+j)/4-(i-j)/2)/2,y=(i-x)/3;
		}
		if(x<=n&&y>=1){
			if(fl){
				swap(x,y);
			}
			ans.eb(x,y);
		}
		j+=8;
	}
	printf("%d\n",ans.size());
	for(auto [i,j]:ans){
		printf("%d %d\n",i,j);
	}
}
signed main(){
	int t=1;
	//	scanf("%d",&t);

		Yorushika();
}
